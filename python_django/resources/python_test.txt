def product(*numbers):
    if len(numbers) == 0:
        raise TypeError("Param is error.");
    count = 1;
    for i in numbers:
        count *= i;
    return count;
print(product(*[1,2,4,6,7]));
print(product(1,2,4,6,7));

def trim(s):
    if len(s) == 0 or s.isspace():
        return '';
    while s[0] == ' ':
        s = s[1:];
    while s[-1] == ' ':
        s = s[:-1];
    return s;
print(trim(""));
print(trim("  dsadad  "));
